```python
"""_scale_detection.py — Automatic scale detection + mesh rescaling.

exports: ScaleResult
         detect_scale_aruco(photos, sparse_cloud_dir, marker_size_mm) -> Optional[ScaleResult]
         detect_scale_ruler(photos, sparse_cloud_dir) -> Optional[ScaleResult]
         apply_scale_to_mesh(mesh, scale_factor) -> trimesh.Trimesh
used_by: lithicore photogrammetry pipeline
rules:   Pure functions, no GUI imports. Scale detection operates on sparse cloud
         + source photos, not dense mesh.
agent:   deepseek-v4-flash | 2026-05-27 | Initial — dataclass + mesh transform
"""

from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path
from typing import Optional

import numpy as np
import trimesh


# ──────────────────────────────────────────────
# Data types
# ──────────────────────────────────────────────

@dataclass
class ScaleResult:
    """Result of automatic scale detection.

    Attributes:
        scale_factor: Multiplier to convert arbitrary units to millimetres.
        method: Detection method used ("aruco" | "ruler" | "manual" | "none").
        confidence: 0.0 (none) to 1.0 (certain).
        detected_length_mm: Physical length inferred from the scene (0 if manual).
        warnings: Any issues encountered during detection.
    """
    scale_factor: float
    method: str
    confidence: float
    detected_length_mm: float = 0.0
    warnings: list[str] = field(default_factory=list)


# ──────────────────────────────────────────────
# Mesh transformation
# ──────────────────────────────────────────────

def apply_scale_to_mesh(
    mesh: trimesh.Trimesh,
    scale_factor: float,
) -> trimesh.Trimesh:
    """Apply a uniform scale factor to all mesh vertices.

    Multiplies every vertex coordinate by scale_factor, preserving
    mesh topology (faces, normals, face normals are recomputed).

    Args:
        mesh: Input mesh in arbitrary coordinates.
        scale_factor: Multiplier (must be > 0).

    Returns:
        New mesh with scaled vertices.

    Raises:
        ValueError if scale_factor <= 0.
    """
    if scale_factor <= 0:
        raise ValueError(
            f"Scale factor must be positive, got {scale_factor}"
        )

    scaled = mesh.copy()
    scaled.vertices = mesh.vertices * scale_factor
    # Update normals since vertex positions changed
    scaled.face_normals = None
    scaled.vertex_normals = None
    return scaled
```

```python
# ──────────────────────────────────────────────
# ArUco marker detection
# ──────────────────────────────────────────────

SUPPORTED_IMAGE_EXTENSIONS = {".jpg", ".jpeg", ".png", ".tiff", ".tif"}
ARUCO_MARKER_SIZE_MM_DEFAULT = 20.0


def _read_photos(photo_dir: Path) -> list[Path]:
    """Return sorted list of supported image files in a directory."""
    return sorted([
        p for p in photo_dir.iterdir()
        if p.suffix.lower() in SUPPORTED_IMAGE_EXTENSIONS
    ])


def detect_scale_aruco(
    photo_dir: Path,
    sparse_dir: Path,
    marker_size_mm: float = ARUCO_MARKER_SIZE_MM_DEFAULT,
) -> Optional[ScaleResult]:
    """Detect ArUco markers in photos and compute scale factor.

    Detects ArUco markers in each photo, triangulates corner positions
    using COLMAP's camera poses (cameras.bin + images.bin), and computes
    the scale factor from the known physical marker size.

    Args:
        photo_dir: Directory containing input photos.
        sparse_dir: Directory containing COLMAP sparse reconstruction
                    (cameras.bin, images.bin, points3D.bin).
        marker_size_mm: Physical size of the ArUco marker in mm.

    Returns:
        ScaleResult if markers detected and triangulated successfully,
        None if no markers found or COLMAP data unavailable.
    """
    try:
        import cv2
        import cv2.aruco as aruco
    except ImportError:
        return ScaleResult(
            scale_factor=1.0,
            method="aruco",
            confidence=0.0,
            detected_length_mm=0.0,
            warnings=["OpenCV not installed. Install with: pip install opencv-python"],
        )

    photos = _read_photos(photo_dir)
    if not photos:
        return None

    # Try to read COLMAP data for camera poses
    try:
        cameras, images = _read_colmap_sparse(sparse_dir)
    except (FileNotFoundError, ValueError):
        # No COLMAP data — try pure 2D ArUco detection for size estimation
        return None

    # Detect markers in each photo
    dictionary = aruco.getPredefinedDictionary(aruco.DICT_6X6_250)
    params = aruco.DetectorParameters()
    detector = aruco.ArucoDetector(dictionary, params)

    # Collect all marker corner observations (image_id, marker_id, corner_idx, pixel_coord)
    marker_observations: dict[int, list[tuple[int, int, np.ndarray]]] = {}
    # marker_id -> [(image_id, corner_idx, pixel_coord)]

    for photo_path in photos:
        img = cv2.imread(str(photo_path), cv2.IMREAD_GRAYSCALE)
        if img is None:
            continue
        corners, ids, _ = detector.detectMarkers(img)
        if ids is None:
            continue
        for marker_idx, marker_id in enumerate(ids.flatten()):
            if marker_id not in marker_observations:
                marker_observations[int(marker_id)] = []
            for corner_idx in range(4):
                pixel = corners[marker_idx][0][corner_idx]
                # Look up which COLMAP image this corresponds to
                for image_id, cam in images.items():
                    if cam.name == photo_path.name:
                        marker_observations[int(marker_id)].append(
                            (image_id, corner_idx, np.array([pixel[0], pixel[1]]))
                        )
                        break

    if not marker_observations:
        return None

    # For each marker with sufficient observations, triangulate corners
    scales: list[float] = []

    for marker_id, observations in marker_observations.items():
        # Group observations by corner index
        corner_groups: dict[int, list[tuple[int, np.ndarray]]] = {}
        for image_id, corner_idx, pixel in observations:
            if corner_idx not in corner_groups:
                corner_groups[corner_idx] = []
            corner_groups[corner_idx].append((image_id, pixel))

        # Triangulate each corner that was seen in ≥2 views
        corners_3d: dict[int, np.ndarray] = {}
        for corner_idx, obs in corner_groups.items():
            if len(obs) < 2:
                continue
            # Triangulate using camera poses
            pts_3d = _triangulate_corner(obs, cameras, images)
            if pts_3d is not None:
                corners_3d[corner_idx] = pts_3d

        if len(corners_3d) < 4:
            continue

        # Compute distances between adjacent corners in 3D
        adjacent_pairs = [(0, 1), (1, 2), (2, 3), (3, 0)]
        for i, j in adjacent_pairs:
            if i in corners_3d and j in corners_3d:
                dist = np.linalg.norm(corners_3d[i] - corners_3d[j])
                if dist > 0:
                    # Physical distance for a marker of marker_size_mm:
                    # the marker has a black border of 1 bit, so 6x6 bits + 2 border bits = 8 units
                    # Physical adjacent corner distance = marker_size_mm
                    scales.append(marker_size_mm / dist)

    if not scales:
        return None

    # Median filter outlier scales
    scales_arr = np.array(scales)
    median = np.median(scales_arr)
    std = scales_arr.std()
    valid = scales_arr[np.abs(scales_arr - median) <= 2 * std]

    if len(valid) == 0:
        return None

    final_scale = float(np.median(valid))
    confidence = min(1.0, len(valid) / 10.0)  # More observations = higher confidence

    return ScaleResult(
        scale_factor=final_scale,
        method="aruco",
        confidence=confidence,
        detected_length_mm=marker_size_mm,
    )


def _read_colmap_sparse(sparse_dir: Path) -> tuple:
    """Read COLMAP sparse reconstruction data.

    Returns (cameras, images) dicts from COLMAP binary format.

    Uses pycolmap if available, otherwise falls back to direct binary parsing.
    """
    try:
        from pycolmap import read_cameras_binary, read_images_binary
        cameras = read_cameras_binary(str(sparse_dir / "cameras.bin"))
        images = read_images_binary(str(sparse_dir / "images.bin"))
        return cameras, images
    except ImportError:
        # Fallback: parse COLMAP binary format directly
        # (Simplified parser — handles standard COLMAP output)
        import struct

        def _read_cameras_bin(path: Path) -> dict:
            """Minimal COLMAP cameras.bin parser."""
            cameras = {}
            with open(path, "rb") as f:
                num_cameras = struct.unpack("Q", f.read(8))[0]
                for _ in range(num_cameras):
                    camera_id = struct.unpack("I", f.read(4))[0]
                    model_id = struct.unpack("i", f.read(4))[0]
                    width = struct.unpack("Q", f.read(8))[0]
                    height = struct.unpack("Q", f.read(8))[0]
                    # Read camera parameters (varies by model_id)
                    num_params = {0: 4, 1: 4, 2: 5, 3: 8, 4: 5, 5: 8, 6: 12, 7: 3}[model_id]
                    params = struct.unpack(f"{num_params}d", f.read(num_params * 8))
                    cameras[camera_id] = type("Camera", (), {
                        "camera_id": camera_id,
                        "model_id": model_id,
                        "width": width,
                        "height": height,
                        "params": params,
                    })()
            return cameras

        def _read_images_bin(path: Path) -> dict:
            """Minimal COLMAP images.bin parser."""
            images = {}
            with open(path, "rb") as f:
                num_images = struct.unpack("Q", f.read(8))[0]
                for _ in range(num_images):
                    image_id = struct.unpack("I", f.read(4))[0]
                    qw, qx, qy, qz = struct.unpack("dddd", f.read(32))
                    tx, ty, tz = struct.unpack("ddd", f.read(24))
                    camera_id = struct.unpack("I", f.read(4))[0]
                    name_bytes = b""
                    while True:
                        byte = f.read(1)
                        if byte == b"\x00":
                            break
                        name_bytes += byte
                    name = name_bytes.decode("utf-8")
                    num_points = struct.unpack("Q", f.read(8))[0]
                    # Skip point2D data
                    f.read(num_points * 24)
                    img = type("Image", (), {
                        "image_id": image_id,
                        "name": name,
                        "qw": qw, "qx": qx, "qy": qy, "qz": qz,
                        "tx": tx, "ty": ty, "tz": tz,
                        "camera_id": camera_id,
                    })()
                    images[image_id] = img
            return images

        cameras = _read_cameras_bin(sparse_dir / "cameras.bin")
        images = _read_images_bin(sparse_dir / "images.bin")
        return cameras, images


def _triangulate_corner(
    observations: list[tuple[int, np.ndarray]],
    cameras: dict,
    images: dict,
) -> Optional[np.ndarray]:
    """Triangulate a 3D point from observations in multiple views.

    Uses the Direct Linear Transform (DLT) on COLMAP camera poses.
    """
    import cv2

    points_2d = []
    projection_matrices = []

    for image_id, pixel in observations[:10]:  # Limit to 10 views
        if image_id not in images or images[image_id].camera_id not in cameras:
            continue

        img = images[image_id]
        cam = cameras[img.camera_id]

        # Build camera matrix from COLMAP intrinsics
        fx, fy, cx, cy = cam.params[0], cam.params[1], cam.params[2], cam.params[3]
        K = np.array([[fx, 0, cx], [0, fy, cy], [0, 0, 1]], dtype=np.float64)

        # Build rotation matrix from quaternion
        qw, qx, qy, qz = img.qw, img.qx, img.qy, img.qz
        R = np.array([
            [1 - 2*qy*qy - 2*qz*qz, 2*qx*qy - 2*qz*qw, 2*qx*qz + 2*qy*qw],
            [2*qx*qy + 2*qz*qw, 1 - 2*qx*qx - 2*qz*qz, 2*qy*qz - 2*qx*qw],
            [2*qx*qz - 2*qy*qw, 2*qy*qz + 2*qx*qw, 1 - 2*qx*qx - 2*qy*qy],
        ], dtype=np.float64)

        # Build projection matrix P = K[R|t]
        t = np.array([[img.tx], [img.ty], [img.tz]], dtype=np.float64)
        Rt = np.hstack([R, t])
        P = K @ Rt

        points_2d.append(pixel)
        projection_matrices.append(P)

    if len(points_2d) < 2:
        return None

    # DLT triangulation
    points_2d = np.array(points_2d, dtype=np.float64)
    X = cv2.triangulatePoints(
        projection_matrices[0],
        projection_matrices[1],
        points_2d[0:1].T,
        points_2d[1:2].T,
    )
    X = X[:3] / X[3]  # Homogeneous to 3D
    return X.flatten()
```

```python
# ──────────────────────────────────────────────
# Ruler/scale bar detection
# ──────────────────────────────────────────────

def detect_scale_ruler(
    photo_dir: Path,
    sparse_dir: Optional[Path],
) -> Optional[ScaleResult]:
    """Detect a standard ruler in photos and compute scale factor.

    Uses adaptive thresholding, Hough line detection, and tick mark
    frequency analysis to locate a standard scale bar and determine
    the pixel-per-mm ratio.

    Args:
        photo_dir: Directory containing input photos.
        sparse_dir: Optional — not used for this method
                    (scale determined from 2D analysis alone).

    Returns:
        ScaleResult if a ruler was detected reliably, None otherwise.
    """
    try:
        import cv2
    except ImportError:
        return ScaleResult(
            scale_factor=1.0,
            method="ruler",
            confidence=0.0,
            detected_length_mm=0.0,
            warnings=["OpenCV not installed. Install with: pip install opencv-python"],
        )

    from scipy.signal import argrelextrema

    photos = _read_photos(photo_dir)
    if not photos:
        return None

    # Select up to 5 photos for analysis
    sample_photos = photos[:5]
    ratios: list[float] = []

    for photo_path in sample_photos:
        img = cv2.imread(str(photo_path), cv2.IMREAD_GRAYSCALE)
        if img is None:
            continue

        # Adaptive threshold to handle varied lighting
        thresh = cv2.adaptiveThreshold(
            img, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
            cv2.THRESH_BINARY_INV, 21, 4
        )

        # Edge detection
        edges = cv2.Canny(thresh, 50, 150)

        # Hough line detection for ruler edges
        lines = cv2.HoughLinesP(
            edges, rho=1, theta=np.pi / 180,
            threshold=100, minLineLength=50, maxLineGap=10
        )

        if lines is None or len(lines) < 2:
            continue

        # Find the longest straight line (likely the ruler edge)
        longest_line = max(lines, key=lambda l: np.linalg.norm(
            [l[0][2] - l[0][0], l[0][3] - l[0][1]]
        ))
        x1, y1, x2, y2 = longest_line[0]

        # Extract pixel intensity profile along the ruler axis
        num_samples = 200
        profile = np.zeros(num_samples)
        for i in range(num_samples):
            t = i / (num_samples - 1)
            x = int(x1 + t * (x2 - x1))
            y = int(y1 + t * (y2 - y1))
            if 0 <= x < img.shape[1] and 0 <= y < img.shape[0]:
                profile[i] = img[y, x]

        # Normalize profile
        profile = (profile - profile.min()) / (profile.max() - profile.min() + 1e-6)

        # Find local minima (tick marks) using argrelextrema
        minima = argrelextrema(profile, np.less, order=5)[0]

        if len(minima) < 3:
            continue

        # Compute median spacing between minima in pixels
        spacings = np.diff(minima) / num_samples * np.linalg.norm([x2 - x1, y2 - y1])
        median_spacing_px = np.median(spacings)

        # Assume 1 mm between tick marks (standard ruler)
        # This gives us pixels per mm
        if median_spacing_px > 0:
            ratios.append(median_spacing_px)

    if not ratios:
        return None

    # Median filter outliers
    ratios_arr = np.array(ratios)
    median = np.median(ratios_arr)
    std = ratios_arr.std()
    valid = ratios_arr[np.abs(ratios_arr - median) <= 2 * std]

    if len(valid) == 0:
        return None

    if len(valid) < 2:
        return None  # Need at least 2 photos for cross-validation

    # pixel_per_mm = median spacing in pixels → this approximates 1mm in pixel space
    # But this is still in 2D — we need to relate it to 3D depth
    # For now, return a low-confidence estimate
    pixel_per_mm = float(np.median(valid))
    confidence = min(0.5, len(valid) * 0.1)  # Lower confidence than ArUco

    # Approximate scale: this is a 2D measurement. In 3D, the scale factor
    # depends on the distance from the camera. We return the 2D estimate
    # and the pipeline will use it as a rough approximation.
    return ScaleResult(
        scale_factor=pixel_per_mm,
        method="ruler",
        confidence=confidence,
        detected_length_mm=1.0,
    )
```
